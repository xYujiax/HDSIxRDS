import matplotlib.pyplot as plt
import matplotlib.ticker as mticker
import seaborn as sns

sns.set_theme(style="whitegrid", palette="muted")
df_plot = df.reset_index()

fig, axes = plt.subplots(2, 3, figsize=(18, 10))
fig.suptitle("Monovariate EDA – California County Uninsured Data (2008–2022)", fontsize=15, fontweight="bold", y=1.01)

# --- 1. Distribution of % uninsured (all county-years) ---
ax = axes[0, 0]
sns.histplot(df_plot["pct_uninsured"], bins=30, kde=True, ax=ax, color="steelblue")
ax.axvline(df_plot["pct_uninsured"].median(), color="crimson", linestyle="--", label=f'Median: {df_plot["pct_uninsured"].median():.1f}%')
ax.set_title("Distribution of % Uninsured\n(all county-years)")
ax.set_xlabel("% Uninsured")
ax.legend()

# --- 2. Statewide average % uninsured over time ---
ax = axes[0, 1]
yearly = df_plot.groupby("year")["pct_uninsured"].mean()
ax.plot(yearly.index, yearly.values, marker="o", color="steelblue", linewidth=2)
ax.fill_between(yearly.index, yearly.values, alpha=0.15, color="steelblue")
ax.set_title("Avg % Uninsured Over Time\n(county mean)")
ax.set_xlabel("Year")
ax.set_ylabel("% Uninsured")
ax.xaxis.set_major_locator(mticker.MultipleLocator(2))

# --- 3. Boxplot of % uninsured by year ---
ax = axes[0, 2]
df_plot.boxplot(column="pct_uninsured", by="year", ax=ax, patch_artist=True,
                boxprops=dict(facecolor="steelblue", alpha=0.5),
                medianprops=dict(color="crimson", linewidth=2))
ax.set_title("% Uninsured by Year\n(county distribution)")
ax.set_xlabel("Year")
ax.set_ylabel("% Uninsured")
fig.texts = [t for t in fig.texts if "Boxplot" not in t.get_text()]  # remove auto title
ax.tick_params(axis="x", rotation=45)

# --- 4. Distribution of county population (log scale) ---
ax = axes[1, 0]
# Use one year to avoid duplicating counties
pop_2022 = df_plot[df_plot["year"] == 2022]["population"]
sns.histplot(pop_2022, bins=20, kde=True, ax=ax, color="teal", log_scale=True)
ax.set_title("County Population Distribution\n(2022, log scale)")
ax.set_xlabel("Population (log scale)")

# --- 5. Top 15 counties by avg % uninsured ---
ax = axes[1, 1]
county_avg = df_plot.groupby("County")["pct_uninsured"].mean().sort_values(ascending=False).head(15)
county_avg.plot(kind="barh", ax=ax, color="steelblue", edgecolor="white")
ax.invert_yaxis()
ax.set_title("Top 15 Counties\nby Avg % Uninsured")
ax.set_xlabel("Avg % Uninsured")
ax.set_ylabel("")

# --- 6. Distribution of margin of error ---
ax = axes[1, 2]
sns.histplot(df_plot["pct_uninsured_moe"], bins=30, kde=True, ax=ax, color="slategray")
ax.axvline(df_plot["pct_uninsured_moe"].median(), color="crimson", linestyle="--",
           label=f'Median: {df_plot["pct_uninsured_moe"].median():.1f}pp')
ax.set_title("Distribution of Estimate MOE\n(% uninsured margin of error)")
ax.set_xlabel("Margin of Error (percentage points)")
ax.legend()

plt.tight_layout()
plt.show()
